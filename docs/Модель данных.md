# Модель данных для приложения «Аптека „Витаминка“»

```java
/* --=== Пользователь ===-- */
User(int userID,      // Идентификатор пользователя
     string login,    // Логин
     string password, // Пароль
     FK Role.roleID,  // Роль
     string name)     // Имя пользователя
     
/* --=== Роль пользователя ===-- */
Role(int roleID,        // Идентификатор роли
     string userStatus) // Роль пользователя (покупатель, кассир, администратор)
     
/* --=== Препарат ===-- */
Drug(int drugID,                     // Идентификатор препарата
     string name,                    // Название препарата
     string category,                // Категория препарата (рецептурное, свободное)
     string description,             // Описание препарата
     sting INN,                      // Международное название препарата
     FK DrugType.typeID,             // Тип препарата
     string dose,                    // Дозировка
     FK Manufacturer.manufacturerID, // Производитель
     int barcode)                    // Штрих-код

/* --=== Производитель ===-- */
Manufacturer(int manufacturerID,   // Идентификатор производителя
             string name,          // Название производителя
             string Country,       // Страна
             string contactNumber) // Контактные данные

/* --=== Тип препарата ===-- */
DrugType(int typeID,   // Идентификатор типа препарата
         string name,  // Тип препарата (сироп, таблетки и т.п)
         int parentID) // Родительский тип (например, таблетки -> оболочечные таблетки)

/* --=== Резервация товара ===-- */
Reservation(int reservationID, // Идентификатор резервации 
            FK Batch.batchID,  // Поставка
            FK Order.orderID,  // Заказ
            int quantity,      // Количество
            date expiresAt)    // Когда завершается

/* --=== Корзина ===-- */
Cart(int cartID,             // Идентификатор корзины
     FK Customer.customerID, // Пользователь
     date createdAt,         // Дата создания
     date updatedAt,         // Дата последнего обновления
     string status)          // active, converted, abandoned

/* --=== Позиция корзины ===-- */
CartItem(int cartItemID, // Идентификатор позиции в корзине
         FK Cart.cartID, // Идентификатор корзины
         FK Drug.drugID, // Идентификатор препарата
         int quantity)   // Кол-во

/* --=== Заказ ===-- */
Order(int orderID,            // Идентификатор заказа
      FK Customer.customerID, // Пользователь
      date createdAt,         // Дата заказа
      int totalAmount,        // Кол-во товара
      string status,          // Статус (принят, оплачен, собирается, отправлен, в ПВЗ, получен)
      string deliveryType)    // Тип доставки (из ПВЗ, курьер)

/* --=== Позиция заказа ===-- */
OrderItem(int orderItemID,  // Идентификатор позиции в заказе
          FK Order.orderID, // Заказ
          FK Batch.batchID, // Номер поставки
          int quantity,     // Кол-во товара
          int price)        // Цена

/* --=== Партия ===-- */
Batch(int batchID,       // Идентификатор партии
      FK Drug.drugID,    // Препарат
      FK Supply.supplyID // Поставка
      int number,        // Номер партии
      date shelfLife,    // Срок годности
      date arrivalDate,  // Дата партии
      int price)         // Цена партии

/* --=== Склад ===-- */
Warehouse(int warehouseID, // Идентификатор склада
          string name,     // Название склада
          string address)  // Адрес склада

/* --=== Остаток товара ===-- */
Stock(int stockID,              // Идентификатор остатка
      FK Warehouse.warehouseID, // Склад
      FK Batch.batchID,         // Поставка
      int amount)               // Оставшееся количество

/* --=== Поставщик ===-- */
Supplier(int supplierID, // Идентификатор поставщика
         string name,    // Название поставщика
         string number)  // Контактный номер

/* --=== Поставка ===-- */
Supply(int supplyID,           // Идентификатор поставки
       FK Supplier.supplierID, // Поставщик
       date supplyDate,        // Дата поставки
       string status)          // Статус поставки (заказана, доставлена, на складе)
```
